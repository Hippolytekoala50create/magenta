        import { initializeApp } from "https://www.gstatic.com/firebasejs/10.7.1/firebase-app.js";
        import { getFirestore, collection, onSnapshot, addDoc, serverTimestamp } from "https://www.gstatic.com/firebasejs/10.7.1/firebase-firestore.js";
        import { getAuth, signInAnonymously } from "https://www.gstatic.com/firebasejs/10.7.1/firebase-auth.js";
        // ============================================================
        // 1. COLLEZ VOS CODES FIREBASE CI-DESSOUS
        // ============================================================
        const firebaseConfig = {
            apiKey: "AIzaSyBYFKpujAoak5TbFCqoKXRc3Nl032-ZH_Y",
            authDomain: "ai-hub-marketplace.firebaseapp.com",
            projectId: "ai-hub-marketplace",
            storageBucket: "ai-hub-marketplace.firebasestorage.app",
            messagingSenderId: "264734345124",
            appId: "1:264734345124:web:2d5472098144d330c4a331" // <--- C'EST ICI QUE VOUS COLLEZ L'APP ID (ex: 1:123456:web:...)
        };
        // ============================================================
        const app = initializeApp(firebaseConfig);
        const db = getFirestore(app);
        const auth = getAuth(app);
        const appId = "ai-hub-global-store";
        let tools = [];
        signInAnonymously(auth);
        const toolsGrid = document.getElementById('tools-grid');
        const modal = document.getElementById('modal');
        onSnapshot(collection(db, 'artifacts', appId, 'public', 'data', 'tools'), (snapshot) => {
            tools = snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() }));
            renderTools();
        }, (error) => {
            console.error("Erreur Firebase:", error);
            toolsGrid.innerHTML = '<div class="col-span-full py-10 text-red-500 text-center">Erreur de connexion : Vérifiez vos clés et vos règles Firestore.</div>';
        });
        function renderTools() {
            if (tools.length === 0) {
                toolsGrid.innerHTML = '<div class="col-span-full py-20 text-center text-gray-400">Aucune IA pour le moment. Soyez le premier !</div>';
                return;
            }
            toolsGrid.innerHTML = tools.map(tool => `
                <div class="bg-white p-5 rounded-[2.5rem] border border-gray-100 shadow-sm hover:shadow-2xl transition-all">
                    <div class="w-14 h-14 bg-blue-600 rounded-2xl flex items-center justify-center text-white mb-4 shadow-lg shadow-blue-100">
                        <i data-lucide="zap"></i>
                    </div>
                    <h3 class="font-bold text-lg text-gray-900">${tool.name}</h3>
                    <p class="text-xs text-blue-500 font-bold uppercase tracking-wider mb-4">${tool.category}</p>
                    <div class="pt-4 border-t border-gray-50 flex items-center justify-between">
                        <span class="text-xs font-black px-3 py-1 bg-gray-100 rounded-full">PREMIUM</span>
                        <div class="flex items-center gap-1 text-orange-500">
                            <i data-lucide="star" class="w-3 h-3 fill-current"></i>
                            <span class="text-xs font-bold">4.9</span>
                        </div>
                    </div>
                </div>
            `).join('');
            lucide.createIcons();
        }
        window.handleFormSubmit = async (e) => {
            e.preventDefault();
            const formData = new FormData(e.target);
            const btn = e.target.querySelector('button');
            btn.innerText = "Validation du paiement...";
            btn.disabled = true;
            try {
                await addDoc(collection(db, 'artifacts', appId, 'public', 'data', 'tools'), {
                    name: formData.get('name'),
                    category: formData.get('category'),
                    createdAt: serverTimestamp()
                });
                alert("Votre IA a été soumise avec succès !");
                window.toggleModal(false);
            } catch (err) {
                alert("Erreur de soumission. Vérifiez vos règles de sécurité Firestore.");
            } finally {
                btn.innerText = "Procéder à la publication";
                btn.disabled = false;
            }
        };
        window.toggleModal = (show) => {
            modal.style.display = show ? 'flex' : 'none';
        };
        lucide.createIcons();
    </script>
